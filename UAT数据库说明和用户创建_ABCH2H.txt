3. 创建表空间
--创建数据表空间（datafile路径需修改）
	create tablespace H2H_USE 
	datafile '$ORACLE_HOME/oracle_tablespaces/H2H_USE_TBSPACE_1.dbf' 
		size 1024M 
		autoextend on 
		next 20M 
		maxsize 20480M,
	'$ORACLE_HOME/oracle_tablespaces/H2H_USE_TBSPACE_2.dbf' 
		size 1024M 
		autoextend on 
		next 20M 
		maxsize 20480M;

--创建索引表空间（datafile路径需修改）
	create tablespace H2H_IDX 
	datafile '$ORACLE_HOME/oracle_tablespaces/H2H_IDX_TBSPACE.dbf' 
	size 512M 
	autoextend on next 5M maxsize 10240M;

4. 用户创建脚本
	(1)h2hset用户
-- Create the user 
	create user h2hset
	  default tablespace H2H_USE
	  temporary tablespace TEMP
	  IDENTIFIED BY h2hset
	  profile DEFAULT;
	  
-- Grant/Revoke role privileges 
	grant connect to h2hset;
	grant resource to h2hset;

-- Grant/Revoke system privileges
	grant create materialized view to h2hset
	grant create sequence to h2hset;
	grant create session to h2hset;
	grant create table to h2hset;
	grant create trigger to h2hset;
	grant create view to h2hset;
	grant unlimited tablespace to h2hset;

	
	(2) 其他用户创建（h2hset创建和用户对象导入完毕后再创建）
	
h2happ用户
-- Create the user 
	create user h2happ
	  default tablespace H2H_USE
	  temporary tablespace TEMP
	  IDENTIFIED BY h2happ
	  profile DEFAULT;
  
-- Grant/Revoke role privileges 
	grant connect to h2happ;
	grant resource to h2happ;
 
---Grant/Revoke object privileges
	all on H2HSET.TABLE 
	all on H2HSET.VIEW 
	all on H2HSET.SEQUENCE 
	all on H2HSET.FUNCTION 

	执行以下sql，获取权限赋值sql（需要dba权限，H2HSET用户）
	select 'grant all on '||owner||'.'||object_name|| ' to H2HAPP;'
	from dba_objects
	where object_type='TABLE' and owner='H2HSET'
	union all
	select 'grant all on '||owner||'.'||object_name|| ' to H2HAPP;'
	from dba_objects
	where object_type='VIEW' and owner='H2HSET'
	union all
	select 'grant all on '||owner||'.'||object_name|| ' to H2HAPP;'
	from dba_objects
	where object_type='SEQUENCE' and owner='H2HSET'
	union all
	select 'grant all on '||owner||'.'||object_name|| ' to H2HAPP;'
	from dba_objects
	where object_type='FUNCTION' and owner='H2HSET';
	执行第1步获取到的全部的权限赋值sql
