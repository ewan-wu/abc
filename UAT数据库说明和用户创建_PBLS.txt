3. 创建表空间
	--1.创建数据表空间（datafile路径需修改）
create tablespace PBLS_USE 
datafile '$ORACLE_HOME/oracle_tablespaces/PBLS_USE_TBSPACE_1.dbf' 
	size 1024M 
	autoextend on 
	next 20M 
	maxsize 20480M,
'$ORACLE_HOME/oracle_tablespaces/PBLS_USE_TBSPACE_2.dbf' 
	size 1024M 
	autoextend on 
	next 20M 
	maxsize 20480M;

--2.创建索引表空间（datafile路径需修改）
create tablespace PBLS_IDX 
datafile '$ORACLE_HOME/oracle_tablespaces/PBLS_IDX_TBSPACE.dbf' 
size 512M 
autoextend on next 5M maxsize 10240M;

4. 用户创建脚本
	(1)pblsset用户
-- Create the user 
	create user pblsset
	  default tablespace PBLS_USE
	  temporary tablespace TEMP
	  IDENTIFIED BY pblsset
	  profile DEFAULT;
  
-- Grant/Revoke role privileges 
	grant connect to pblsset;
	grant resource to pblsset;

-- Grant/Revoke system privileges
	grant create materialized view to pblsset
	grant create sequence to pblsset;
	grant create session to pblsset;
	grant create table to pblsset;
	grant create trigger to pblsset;
	grant create view to pblsset;
	grant unlimited tablespace to pblsset;

(2) 其他用户创建（pblsset创建和用户对象导入完毕后再创建） 
pblsapp用户创建
-- Create the user 
	create user pblsapp
	  default tablespace PBLS_USE
	  temporary tablespace TEMP
	  IDENTIFIED BY pblsapp
	  profile DEFAULT;
  
-- Grant/Revoke role privileges 
	grant connect to pblsapp;
	grant resource to pblsapp;

---Grant/Revoke object privileges
	all on PBLSSET.TABLE
	all on PBLSSET.VIEW
	all on PBLSSET.SEQUENCE
	all on PBLSSET.FUNCTION

	执行以下sql，获取权限赋值sql（需要dba权限，PBLSSET用户）
	select 'grant all on '||owner||'.'||object_name|| ' to PBLSAPP;'
	from dba_objects
	where object_type='TABLE' and owner='PBLSSET'
	union all
	select 'grant all on '||owner||'.'||object_name|| ' to PBLSAPP;'
	from dba_objects
	where object_type='VIEW' and owner='PBLSSET'
	union all
	select 'grant all on '||owner||'.'||object_name|| ' to PBLSAPP;'
	from dba_objects
	where object_type='SEQUENCE' and owner='PBLSSET'
	union all
	select 'grant all on '||owner||'.'||object_name|| ' to PBLSAPP;'
	from dba_objects
	where object_type='FUNCTION' and owner='PBLSSET';
	执行第1步获取到的全部的权限赋值sql
